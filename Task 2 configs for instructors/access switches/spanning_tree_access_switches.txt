This is a re-usable config for RSTP on the exos access switches. 


disable stpd s0
configure stpd s0 mode dot1w
enable stpd s0


# add each one to each respective access switch

# note only need to add port 11 on clinical/radiology since they are using the LAG config.

config stpd s0 add vlan Admin add ports 12
config stpd s0 add vlan Radiology add ports 11
config stpd s0 add vlan Clinical add ports 11
config stpd s0 add vlan Guest add ports 12
config stpd s0 add vlan Radiology add ports 11
